column_header = diabetesDataFrame.columns.tolist()
column_header.remove('Outcome')
df = diabetesDataFrame.copy()

for i in column_header :
    # نرمالیزه کردن
    df[i] = (df[i]-df[i].mean())/df[i].std()

    # میانگین و انحراف معیار افراد دیابتی
    Diabetes_mean = df[df['Outcome'] == 1][i].mean()
    Diabetes_std  = df[df['Outcome'] == 1][i].std()

    # میانگین و انحراف معیار افراد غیر دیابتی
    nonDiabetes_mean = df[df['Outcome'] == 0][i].mean()
    nonDiabetes_std  = df[df['Outcome'] == 0][i].std()

    # بدست آوردن تابع چگالی احتمال، برای هر یک از گروه ها
    x = np.linspace(-4, 4, 1000)
    y_Diabetes = norm.pdf(x, Diabetes_mean, Diabetes_std)
    y_nonDiabetes = norm.pdf(x, nonDiabetes_mean, nonDiabetes_std)
    
    # رسم نمودار
    plt.plot(x, y_Diabetes, label = 'Diabetes')
    plt.plot(x, y_nonDiabetes, label = 'nonDiabetes')
    #plt.scatter(x, y_Diabetes)
    #plt.hexbin(x, y_Diabetes)
    plt.title(i)
    plt.grid()
    plt.legend()
    plt.show()

سوال اول:¶

در ادامه برای حل مشکل مقادیر از دست رفته، دو راه در پیش داریم. یکی اینکه کل ستون را حذف کنیم که در این صورت ممکن است آن ستون حاوی اطلاعات خوبی باشد که با اینکار آنها را نیز از دست میدهیم. راه دوم این است که مقادیر خالی را با آماره ها برای مثال مد یا میانگین پر کنیم. در این صورت هیچ دیتایی از بین نمیرود اما حالت بدی که ممکن است رخ دهد این است که اطلاعات زیادی خالی باشد و ما حجم زیادی را داده سازی کنیم و دیتای بی معنی زیادی تولید کنیم

سوال سوم:¶

دو متد عددی نرمالایز و استانداردیزیشن جهت پردازش ساده تر اطلاعات انجام میشود در متد نرمالیزیشن متغیر ها به عددی بین صفر و یک تبدیل میشوند اما در استانداردیزیشن متغیرها به شکلی تغییر میکنند که میانگین آنها صفر و انحراف معیار یک داشته باشند با توجه به اینکه داده ها تقریبا توزیع نرمال دارند از متد استانداردیزیشن استفاده میکنیم. بنابراین با اینکار دیتاهای ما تبدیل به توزیع نرمال استاندارد میشوند و مقادیری بین یک و منفی یک پیدا میکنند

df = diabetesDataFrame.copy()

for i in column_header:
    # find mode to replace the garbage values
    mode = stats.mode(diabetesDataFrame[i])
    
    # if any data was less than zero, replace it with mode value
    df[df[i] < 0] = float(mode[0])
        
    # if number of mode > 15% total Size, using mode to replace the NaN values. otherwise, replace it with mean
    if float(mode[1]) > 0.15 * diabetesDataFrame.shape[0]:
        df[i] = df[i].fillna(float(mode[0]))
    else: df[i] = df[i].fillna(df[i].mean())
        
           
    # نرمالیزه کردن
    df[i] = (df[i]-df[i].mean())/df[i].std()

    # میانگین و انحراف معیار افراد دیابتی

سوال سوم: (آموزش، ارزیابی و تنظیم)¶

underfitting:¶

این حالت زمانی اتفاق میوفتد که ما به دلالیل مختلف مانند کم بودن دیتای ترین، بد انتخاب کردن هایپر پارامترها و... نمیتوانیم مدل خود را به خوبی آموزش دهیم و در نتیجه نمیتوانیم به خوبی پیش بینی کنیم و دقت مدل ما کاهش میابد

overfitting:¶

این حالت هنگامی رخ میدهد که ما بیش از حد مدل خود را آموزش میدهیم و در نتیجه حتی نویزها هم مدل میشوند و دقت مدل ما کاهش میابد
