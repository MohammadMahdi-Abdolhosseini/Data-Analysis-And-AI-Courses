    # نرمالیزه کردن
    df_copy[i] = (df_copy[i]-df_copy[i].mean())/df_copy[i].std()

    # میانگین و انحراف معیار افراد زنده
    Alive_mean = df_copy[df_copy['Survived'] == 1][i].mean()
    Alive_std = df_copy[df_copy['Survived'] == 1][i].std()

    # میانگین و انحراف معیار افراد مرده
    Dead_mean = df_copy[df_copy['Survived'] == 0][i].mean()
    Dead_std = df_copy[df_copy['Survived'] == 0][i].std()

    # بدست آوردن تابع چگالی احتمال، برای هر یک از گروه ها
    x = np.linspace(-5,5,100)
    y_Alive = norm.pdf(x, Alive_mean, Alive_std)
    y_Dead = norm.pdf(x, Dead_mean, Dead_std)
    
    # رسم نمودار
    plt.plot(x, y_Alive, label = 'Alive')
    plt.plot(x, y_Dead, label = 'Dead')
    plt.title(i)
    plt.grid()
    plt.legend()
    plt.show()
    
    # بدست آوردن نقاط برخورد دو نمودار و حذف مقادیر ابتدا و انتهایی
    dif =  y_Alive - y_Dead
    dif[:25] = 1
    dif[75:] = 1
    
    if i == 'Fare' :
        index = np.argmin(np.abs(dif))
        Fare_threshold_1 = x[index]

        index = np.argmin(np.abs(dif))
